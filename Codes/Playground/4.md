// Main.hpp
```
#include <iostream>

class TicTacToe
{
	int SIZE;
	char box[3][3];
public:
	TicTacToe(int _size);
	void setVal();
	char getVal(int row, int col);
	int checkCross();
	int checkZero();
	void printTable();
};

TicTacToe::TicTacToe(int _size)
{
    SIZE = _size;
	//Everytime the box will be emptied for a new game. 
	for (int i = 0; i < SIZE; ++i)
	{
		for (int j = 0; j < SIZE; ++j)
		{
			box[i][j] = '_';
		}
	}
}

void TicTacToe::setVal()
{
    int row, col, _val;
    /*row = 0;
    col = 0; 
    _val = ' ';*/
    
    for (int i=0;i<SIZE*SIZE; i++)
    {
        std::cin>> row; 
        std::cin>> col;
        std::cin >> _val;
    	box[row][col] = _val;
    }
}

char TicTacToe::getVal(int row, int col)
{
	return box[row][col];
}

int TicTacToe::checkCross()
{
	int count = 0;
	for (int i=0; i<SIZE; i++)
	{
		for (int j=0; j<SIZE; j++)
		{
			if (box[i][j] == 'x')
			{
				count++;
			} else if (box[j][i] == 'x')
			{
				count++;
			}
			
		}
		if (count == 4)
		{
		    return 1;
		}

	}
	return 0;
}

int TicTacToe::checkZero()
{
	int count = 0;
	for (int i=0; i<SIZE; i++)
	{
		for (int j=0; j<SIZE; j++)
		{
			if (box[i][j] == 'o')
			{
				count++;
			} else if (box[j][i] == 'o')
			{
				count++;
			}
			
		}
		if (count == 4)
		{
		    return 1;
		}
	}
	return 0;
}

void TicTacToe::printTable()
{
	std::cout<<"The table is displayed below: \n";
	for (int i=0; i< SIZE; i++)
	{
	    std::cout<<"|";
		for (int j=0; j< SIZE; j++)
		{
			std::cout<<box[i][j]<<"|";
		}
		std::cout<<"\n";
	}
}
```
